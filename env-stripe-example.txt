# =============================================================================
# STRIPE – нужни .env променливи за абонаменти (копирай в .env)
# =============================================================================
# След като създадеш продукти/цени в Stripe Dashboard, вземи Price ID (price_xxx)
# и го постави отдолу. Ключовете по-долу вече ги имаш в .env.
# =============================================================================

# --- Ключове (задължителни) ---
# STRIPE_SECRET_KEY=sk_test_...   или STRIPE_SECRET_KEY_FIXED=sk_test_...
# STRIPE_WEBHOOK_SECRET=whsec_...  от Stripe Dashboard > Webhooks > Signing secret
# STRIPE_PUBLISH_KEY=pk_test_...   за клиент (по избор)

# --- Price IDs по планове (задължителни за checkout) ---
# ВАЖНО: Трябва Price ID (започва с price_...), НЕ Product ID (prod_...)!
# В Stripe: Products > отвори продукта > под него са "Prices" > копирай Price ID (price_xxx).

# Стартер – препоръчани цени: 4.99 €/месец, 49 €/год. (2 месеца безплатно)
STRIPE_STARTER_MONTHLY_PRICE_ID=price_xxxxx
STRIPE_STARTER_YEARLY_PRICE_ID=price_xxxxx

# Про – препоръчани цени: 8.99 €/месец, 89 €/год. (2 месеца безплатно)
STRIPE_PRO_MONTHLY_PRICE_ID=price_xxxxx
STRIPE_PRO_YEARLY_PRICE_ID=price_xxxxx
# Алтернатива: единичен Price ID за месечен (ако използваш само един)
# STRIPE_PRO_PRICE_ID=price_xxxxx

# Бизнес – препоръчани цени: 19.99 €/месец, 199 €/год. (2 месеца безплатно)
STRIPE_BUSINESS_MONTHLY_PRICE_ID=price_xxxxx
STRIPE_BUSINESS_YEARLY_PRICE_ID=price_xxxxx
# Алтернатива (поддържа се и с печатна грешка):
# STRIPE_BUISNESS_PRICE_ID=price_xxxxx
# STRIPE_BUISNESS_YEARLY_PRICE_ID=price_xxxxx

#  Бизнес    | 19.99 €  | 199 €     | EUR
